テーブル操作関連
********************
postgres           *
********************

・接続
psql -h hostname -U username -d databasename

 ロカール接続
psql -U username

・切断
\q

・データベース作成
create database database_name;

・データベース一覧
¥l

・データベース接続
\c databasename
・ユーザ(ロール)作成
CREATE ROLE role_name WITH LOGIN PASSWORD 'password'
	option
		SUPERUSER
		CREATEDB
		CREATEROLE
・暗号化モジュール作成
CREATE EXTENSION pgcrypto

・権限追加
GRANT 権限 ON 対象 TO 誰に
GRANT SELECT, UPDATE, INSERT ON table_name TO role_name;
・ユーザ情報確認
\du
・権限情報確認
\z

・列の追加
ALTER TABLE products ADD COLUMN description text;

・列の削除
alter table tablename drop columnName cascade;

・制約追加
ALTER TABLE products ADD CONSTRAINT some_name UNIQUE (product_no);
ALTER TABLE products ADD FOREIGN KEY (product_group_id) REFERENCES product_groups;

・制約の削除
ALTER TABLE products DROP CONSTRAINT some_name;
・dump
pg_dump -h hostname -p port -U username -d databasename -f fileName

・リストア
psql -h hostname -p port -U username -d databasename < fileName

・暗号化モジュールインストール
	\dn スキーマ一覧
	create extension pgcrypto with schema public;
	\dx インストール済み拡張一覧
	
********************
MySQL              *
********************
・接続
mysql -u root -p
・切断
quit
・データベース選択
USE db_name;
・ユーザ作成
CREATE USER user IDENTIFIED BY [PASSWORD] 'password';
	※'username'@'hostname'を指定
・権限追加
GRANT 権限 ON レベル TO user;
GRANT ALL ON *.* TO user;

データベースレベル:
GRANT 権限 ON db_name.* TO user;
